\chapter{Observational equality}

\section{Observational equality on $\N$}

\begin{defn}
A \define{(type valued) binary relation} on a type $A$ in context $\Gamma$ is a type family
\begin{equation*}
\Gamma,x:A,y:A\vdash R(x,y)~\mathrm{type}.
\end{equation*}
A \define{small binary relation} on a type $A$ is a term of type
\begin{equation*}
A\to (A\to\UU)
\end{equation*}
\end{defn}
