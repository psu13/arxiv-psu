\subsection{The type of \texorpdfstring{$2$}{2}-element sets}

\begin{thm}
  The type
  \begin{equation*}
    \sm{X:\UU_\bool}X
  \end{equation*}
  is contractible.
\end{thm}

\begin{cor}
  For any $2$-element type $X$, the map
  \begin{equation*}
    (\bool=X)\to X
  \end{equation*}
  given by $p\mapsto\tr_{\mathcal{T}}(p,\btrue)$ is an equivalence.
\end{cor}
