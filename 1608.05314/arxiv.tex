% ******************************************************************
% ** Description:      These macros make it easier to prepare a 
% **                   paper for publication in the archive.
% **                   Just copy header.tex, footer.tex and 
% **                   macros.tex from common to the directory that
% **                   contains the paper to be archived and flick
% ++                   an if switch!      
% ** Author:           Emily Riehl and Dominic Verity
% ** Commenced:        13/10/2015
% ******************************************************************

\newif\ifarxiv

\def\inputarx#1{%
  \ifarxiv
    \input{#1}
  \else
    \input{../common/#1}
  \fi}

\def\bibliographyarx#1{
  \ifarxiv
    \bibliography{#1}
  \else
    \bibliography{../common/#1}
  \fi
}

\arxivfalse
